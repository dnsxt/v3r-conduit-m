// V3R CONDUIT — Extended Doc Converter
// Converts .txt, .py, .json, and Dockerfile files to Google Docs format
// Scoped to V3R/ folder hierarchy only
// Maintains txt/Doc pairs for Drive tool readability

function convertV3RFilesToDocs() {
  var v3rFolderName = "V3R";
  var rootFolders = DriveApp.getFoldersByName(v3rFolderName);

  if (!rootFolders.hasNext()) {
    Logger.log("ERROR: V3R folder not found in My Drive.");
    return;
  }

  var v3rFolder = rootFolders.next();
  Logger.log("Starting conversion in V3R folder hierarchy...");
  processFolder(v3rFolder);
  Logger.log("Conversion complete.");
}

function processFolder(folder) {
  var files = folder.getFiles();
  while (files.hasNext()) {
    var file = files.next();
    var fileName = file.getName();
    var mimeType = file.getMimeType();

    // Skip Google Docs and other Google native formats
    if (mimeType === MimeType.GOOGLE_DOCS ||
        mimeType === MimeType.GOOGLE_SHEETS ||
        mimeType === MimeType.GOOGLE_SLIDES) {
      continue;
    }

    var shouldConvert = false;
    var nameWithoutExtension = fileName;

    if (fileName.toLowerCase().endsWith('.txt')) {
      nameWithoutExtension = fileName.replace(/\.txt$/i, "");
      shouldConvert = true;
    } else if (fileName.toLowerCase().endsWith('.py')) {
      nameWithoutExtension = fileName.replace(/\.py$/i, "");
      shouldConvert = true;
    } else if (fileName.toLowerCase().endsWith('.json')) {
      nameWithoutExtension = fileName.replace(/\.json$/i, "");
      shouldConvert = true;
    } else if (fileName.toLowerCase().endsWith('.ps1')) {
      nameWithoutExtension = fileName.replace(/\.ps1$/i, "");
      shouldConvert = true;
    } else if (fileName.toLowerCase().startsWith('dockerfile')) {
      nameWithoutExtension = fileName;
      shouldConvert = true;
    }

    if (!shouldConvert) {
      continue;
    }

    if (alreadyExists(nameWithoutExtension, folder)) {
      Logger.log('Skipped (Already exists): ' + nameWithoutExtension);
      continue;
    }

    var resource = {
      title: nameWithoutExtension,
      mimeType: MimeType.GOOGLE_DOCS,
      parents: [{id: folder.getId()}]
    };

    try {
      Drive.Files.copy(resource, file.getId(), {convert: true});
      Logger.log('Converted: ' + fileName + ' -> ' + nameWithoutExtension);
    } catch (e) {
      Logger.log('Failed to convert ' + fileName + ': ' + e.toString());
    }
  }

  // Recurse into subfolders
  var subfolders = folder.getFolders();
  while (subfolders.hasNext()) {
    processFolder(subfolders.next());
  }
}

function alreadyExists(name, folder) {

  }

}
